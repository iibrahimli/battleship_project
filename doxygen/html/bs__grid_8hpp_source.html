254  bool place_ship(ship_type type, size_t row, size_t col, ship_orientation orient){
255 
256  // check if ship type TYPE exists
257  if(n_ships.find(type) == n_ships.end()){
258  throw illegal_move_exception("Unknown ship type");
259  }
260 
261  // check if maximum # of ships of type type have already been placed
262  if(n_ships[type] == max_n_ships[type]){
263  throw illegal_move_exception("Ship type has been placed MAX times already");
264  }
265 
266  // holds cells to place ship on if placement is possible
267  std::vector<std::pair<size_t, size_t>> to_place;
268  size_t r = row, c = col;
269 
270  // loop over the ship length and add cells to vector for later placement
271  for(int sz=0; sz<type; ++sz){
272 
273  try{
274  // if any of the cells are not available, the placement cannot be done
275  if(!cell_at(r, c).can_place()) return false;
276  }
277  catch(index_exception& e){
278  // placement is not possible if a coordinate is out of bounds
279  return false;
280  }
281 
282  // if the cell is okay to place, add it to vector
283  to_place.push_back({r, c});
284 
285  // increment column or row value according to orientation
286  if(orient == SO_HOR) ++c;
287  else ++r;
288  }
289 
290  // place ship
291  for(auto& coord : to_place){
292  cell_at(coord.first, coord.second).state = CS_FULL;
293  cell_at(coord.first, coord.second).ship_id = cur_ship_id;
294  }
295 
296  // there is one more ship of type TYPE now
297  ++n_ships[type];
298  ++alive_ships;
299 
300  // next id
301  ++cur_ship_id;
302 
303  // check all ships and set state to ready if all ships are placed
304  bool rd = true;
305  for(auto& tp : n_ships)
306  rd = rd && (tp.second == max_n_ships[tp.first]);
307  if(rd) state = GS_READY;
308 
309  return true;
310  }

322  std::pair<shot_result, int> shoot_at(size_t row, size_t col){
323 
324  // the move is illegal if the cell has been shot before
325  // cell_at() may also throw index_exception
326  if(!cell_at(row, col).can_shoot())
327  throw illegal_move_exception("Cell has been shot before");
328 
329  shot_result sr;
330  int shot_ship_id = -1;
331 
332  // change state of the cell depending on previous state
333  if(cell_at(row, col).state == CS_EMPTY){
334  cell_at(row, col).state = CS_MISSED;
335  sr = SR_MISS;
336  }
337  else if(cell_at(row, col).state == CS_FULL){
338  cell_at(row, col).state = CS_DESTROYED;
339  sr = SR_HIT;
340  shot_ship_id = cell_at(row, col).ship_id;
341 
342  // if ship is sunk
343  if(ship_sunk(shot_ship_id)){
344  sr = SR_SINK;
345  --alive_ships;
346  }
347  }
348 
349  return {sr, shot_ship_id};
350  }
